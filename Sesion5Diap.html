🐍 LIMPIEZA EN PYTHON (COLAB)

# Importar librerías esenciales import pandas as pd import numpy as np import matplotlib.pyplot as plt import seaborn as sns # Cargar dataset df = pd.read_csv('dataset_sucio.csv') # Exploración inicial print("Información del dataset:") print(df.info()) print("\nValores faltantes por columna:") print(df.isnull().sum()) print("\nDuplicados:", df.duplicated().sum()) # Limpieza básica # 1. Eliminar duplicados df_clean = df.drop_duplicates() # 2. Limpiar nombres de columnas df_clean.columns = df_clean.columns.str.strip().str.lower().str.replace(' ', '_') # 3. Manejar valores faltantes # Para numéricos: llenar con mediana numeric_cols = df_clean.select_dtypes(include=[np.number]).columns df_clean[numeric_cols] = df_clean[numeric_cols].fillna(df_clean[numeric_cols].median()) # Para categóricos: llenar con moda categorical_cols = df_clean.select_dtypes(include=['object']).columns for col in categorical_cols: mode_value = df_clean[col].mode()[0] if not df_clean[col].mode().empty else 'Unknown' df_clean[col] = df_clean[col].fillna(mode_value) # Verificar limpieza print("Después de limpieza:") print("Valores faltantes:", df_clean.isnull().sum().sum()) print("Shape original:", df.shape, "→ Shape limpio:", df_clean.shape)

📊 LIMPIEZA EN R (POSIT CLOUD)

# Cargar librerías library(dplyr) library(tidyr) library(VIM) library(ggplot2) # Cargar datos df <- read.csv("dataset_sucio.csv", stringsAsFactors = FALSE) # Exploración inicial cat("Dimensiones del dataset:", dim(df), "\n") summary(df) cat("Valores faltantes por columna:\n") sapply(df, function(x) sum(is.na(x))) # Visualizar patrones de datos faltantes VIM::aggr(df, col = c('navyblue','red'), numbers = TRUE, sortVars = TRUE) # Limpieza con dplyr df_clean <- df %>% # Eliminar duplicados distinct() %>% # Limpiar nombres de columnas rename_all(~ gsub(" ", "_", tolower(trimws(.)))) %>% # Manejar outliers con IQR mutate(across(where(is.numeric), ~ ifelse( abs(. - median(., na.rm = TRUE)) > 3 * IQR(., na.rm = TRUE), median(., na.rm = TRUE), . ))) %>% # Imputar valores faltantes mutate(across(where(is.numeric), ~ ifelse(is.na(.), median(., na.rm = TRUE), .))) %>% mutate(across(where(is.character), ~ ifelse(is.na(.), names(sort(table(.), decreasing = TRUE))[1], .))) # Reporte de limpieza cat("Datos originales:", nrow(df), "filas,", ncol(df), "columnas\n") cat("Datos limpios:", nrow(df_clean), "filas,", ncol(df_clean), "columnas\n") cat("Valores faltantes eliminados:", sum(is.na(df)) - sum(is.na(df_clean)), "\n")

📈 Fortaleza de R

Análisis estadístico robusto + Visualización avanzada + Tidyverse

🎯 MANEJO DE OUTLIERS

📊🚨🔍

📊 MÉTODO IQR

Regla: Q1 - 1.5×IQR ≤ valor ≤ Q3 + 1.5×IQR

✅ Robusto estadísticamente

✅ No asume distribución normal

⚠️ Puede ser muy conservador

📈 Z-SCORE

Regla: |z| > 3 es outlier

✅ Fácil de interpretar

✅ Considera media y desviación

⚠️ Sensible a distribución

🤖 ISOLATION FOREST

Método: Machine Learning

✅ Multivariable

✅ No paramétrico

⚠️ Más complejo de interpretar

🎯 Estrategias de Tratamiento

¿Eliminar, transformar, mantener o investigar?

❓ ESTRATEGIAS PARA DATOS FALTANTES

🎲 MCAR

Missing Completely at Random

Los datos faltantes no dependen de ninguna variable

✅ Seguro eliminar filas

✅ No introduce sesgo

🎯 MAR

Missing at Random

Dependen de variables observadas

✅ Imputación por grupos

✅ Regresión múltiple

🚨 MNAR

Missing Not at Random

El valor faltante ES información

⚠️ Requiere dominio específico

⚠️ Crear categoría "Desconocido"

🛠️ Técnicas de Imputación

Desde simple hasta avanzada: media → KNN → MICE → Deep Learning

✅ MÉTRICAS DE CALIDAD

📊 COMPLETITUD

% de datos no faltantes

Objetivo: > 95%

completitud = (1 - df.isnull().sum() / len(df)) * 100

🎯 CONSISTENCIA

Formatos estandarizados

Fechas, números, categorías

# Verificar formatos de fecha pd.to_datetime(df['fecha'], errors='coerce').isnull().sum()

🔍 PRECISIÓN

Valores dentro de rangos esperados

Edades 0-120, salarios > 0

# Verificar rangos lógicos invalid_ages = df[(df['edad'] < 0) | (df['edad'] > 120)]

🚫 DUPLICACIÓN

% de registros únicos

Objetivo: < 5% duplicados

duplicacion = (df.duplicated().sum() / len(df)) * 100
